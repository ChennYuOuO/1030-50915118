Day.js 日期處理功能展示

Day.js 官方網站

範例程式碼


            <!-- Day.js JS -->
            <script src="https://cdnjs.cloudflare.com/ajax/libs/dayjs/1.11.6/dayjs.min.js"></script>

            <!-- 格式化日期 -->
            dayjs().format('YYYY-MM-DD')

            <!-- 格式化時間 -->
            dayjs().format('HH:mm:ss')

            <!-- 日期加減 -->
            dayjs().add(1, 'day').format('YYYY-MM-DD')
            dayjs().subtract(1, 'month').format('YYYY-MM-DD')
