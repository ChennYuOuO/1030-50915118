範例程式碼


            <!-- 引入 Anime.js -->
            <script src="https://cdn.jsdelivr.net/npm/animejs/lib/anime.min.js"></script>

            <!-- 重複播放動畫示例 -->
            anime({
                targets: '.box',
                translateX: 250,
                rotate: '1turn',
                backgroundColor: '#f39c12',
                duration: 2000,
            });
