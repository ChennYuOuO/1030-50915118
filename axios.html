Axios API 請求範例

Axios 官方網站

範例程式碼


            <!-- 引入 Axios -->
            <script src="https://cdn.jsdelivr.net/npm/axios/dist/axios.min.js"></script>

            <!-- 使用 Axios 發送 GET 請求 -->
            axios.get('https://jsonplaceholder.typicode.com/posts')
                .then(response => {
                    console.log(response.data);
                })
                .catch(error => {
                    console.error(error);
                });
